using EFCoreDemo.NW;

// =================================================================

NorthwindContext context = new();

// =================================================================

void GetAllCategories() {
  foreach (var c in context.Categories) {
    Console.WriteLine($"{c.CategoryId}\t{c.CategoryName}\t{c.Description}");
  }
  Console.WriteLine("==============================");
}

// =================================================================

void GetAllCategoriesByFirstCharacterLambda(string starts) {
  var qry = context.Categories
    .Where(c => c.CategoryName.StartsWith(starts));

  foreach (var item in qry) {
    Console.WriteLine($"{item.CategoryId}\t{item.CategoryName}\t{item.Description}");
  }
}

// =================================================================

void GetAllCategoriesByFirstCharacterQuerySyntax(string starts) {
  var qry = from c in context.Categories
    where c.CategoryName.StartsWith(starts)
    select c;

  foreach (var item in qry) {
    Console.WriteLine($"{item.CategoryId}\t{item.CategoryName}\t{item.Description}");
  }
}

// =================================================================

void GetAllCategoriesDescendingIdOrder() {
  var qry = context.Categories
    .Select (c => new {c.CategoryId, c.CategoryName, c.Description })
    .OrderByDescending(c => c.CategoryId);

  foreach (var item in qry) {
    Console.WriteLine($"{item.CategoryId}\t{item.CategoryName}\t{item.Description}");
  }
}

// =================================================================

void GetAllCategoriesSpecificColumns() {
  var qry = from c in context.Categories
    select new { c.CategoryId, c.CategoryName };

  foreach (var item in qry) {
    Console.WriteLine($"{item.CategoryId}\t{item.CategoryName}");
  }
}

// =================================================================

void GetProductsByCategory() {
  var qry = context.Products
    .Select(p => new { p.ProductId, p.ProductName, p.Category.CategoryName });

  foreach (var item in qry) {
    Console.WriteLine($"{item.ProductId}\t{item.ProductName}\t{item.CategoryName}");
  }
}

// =================================================================

void GetProductsByCategoryAlias() {
  var qry = context.Products
    .Select(p => new {
      ID = p.ProductId,
      Name = p.ProductName,
      Category = p.Category.CategoryName
    });

  foreach (var item in qry) {
    Console.WriteLine($"{item.ID}\t{item.Name}\t{item.Category}");
  }
}

// =================================================================

void GetProductCountByCategory() {
   var qry = context.Products
      .Include (c => c.Category)
      .GroupBy (p => p.Category.CategoryName)
      .Select (g => new {
         Name = g.Key,
         Count = g.Count ()
      })
      .OrderByDescending(cp => cp.Count);

    Console.WriteLine(qry.ToQueryString());

   Console.WriteLine("==========================");

   foreach (var item in qry) {
      Console.WriteLine($"{item.Name}\t{item.Count}");
   }
}

// =================================================================

void InsertCategory(string name, string desc) {
  var newCategory = new Category() {
    CategoryName = name,
    Description = desc
  };

  context.Categories.Add(newCategory);
  context.SaveChanges();
}

void UpdateCategory(int id, string name, string desc) {
  var categoryToUpdate = context.Categories
    .Find(id);

  categoryToUpdate.CategoryName = name;
  categoryToUpdate.Description = desc;

  context.SaveChanges();
}

void DeleteCategory(int id) {
  var categoryToDelete = context.Categories
    .Find(id);

  context.Categories.Remove(categoryToDelete);
  context.SaveChanges();
}

  var pDesc = new SqlParameter("@Description", desc);
