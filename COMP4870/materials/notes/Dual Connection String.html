var connectionString = Environment.GetEnvironmentVariable("ConnectionStrings__bevdb");
if (connectionString.IsNullOrEmpty()) {
    Console.WriteLine("Connection string 'bevdb' not found. Using default connection string.");
    connectionString = builder.Configuration.GetConnectionString("DefaultConnection");
    if (connectionString.IsNullOrEmpty()) {
        connectionString = "NOT FOUND";
        throw new InvalidOperationException("Connection string 'bevdb' and 'DefaultConnection' not found.");
    }
} 

Console.WriteLine($"Connection string: {connectionString}");

builder.Services.AddDbContext<BevDbContext>(options =>
    options.UseSqlServer(connectionString));
