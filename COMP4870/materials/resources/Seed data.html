protected override void Seed(CodeFirstPrimer.Models.NHL.NhlContext context)
        {
            context.Teams.AddOrUpdate(
              t => t.TeamName, getTeams().ToArray());

            context.SaveChanges();

            context.Players.AddOrUpdate(
             p => new { p.JerseyNumber, p.FirstName, p.LastName },
             getPlayers(context).ToArray());

            context.SaveChanges();
        }
