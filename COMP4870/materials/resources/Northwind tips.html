public IActionResult ProductsByCategory () {
   var query = _context.Products
      .Include(c => c.Category)
      .GroupBy (p => p.Category.CategoryName)
      .Select (g => new {
         Name = g.Key,
         Count = g.Count ()
      })
      .OrderByDescending( cp => cp.Count)
      .ToList<object> ();;

   ViewBag.CategoryProduct = query;

   return View ();
}
